#LeNet-5 implementation
from tensorflow.contrib.layers import flatten

def LeNet(x):    
    # Hyperparameters
    mu = 0
    sigma = 0.1
    
    W1 = tf.Variable(tf.truncated_normal(shape=(5,5,3,6), mean = mu, stddev=sigma))
    B1 = tf.Variable(tf.zeros(6))
    strides = [1,1,1,1]
    padding = 'VALID'
    conv1 = tf.nn.conv2d(x, W1, strides, padding) + B1
    
    conv1 = tf.nn.relu(conv1)

    conv1 = tf.nn.max_pool(conv1, ksize=[1,2,2,1], strides=[1,2,2,1], padding='VALID')

    W2 = tf.Variable(tf.truncated_normal(shape=(5,5,6,16), mean = mu, stddev=sigma))
    B2 = tf.Variable(tf.zeros(16))
    padding = 'VALID'
    strides = [1,1,1,1]
    conv2 = tf.nn.conv2d(conv1, W2, strides, padding) + B2
    
    conv2 = tf.nn.relu(conv2)

    conv2 = tf.nn.max_pool(conv2, ksize=[1,2,2,1], strides=[1,2,2,1], padding='VALID')
    conv2 = tf.nn.dropout(conv2, 0.5)

    flatten_layer = flatten(conv2)
    
    w3 = tf.Variable(tf.truncated_normal(shape=(400,120), mean = mu, stddev=sigma))
    b3 = tf.Variable(tf.zeros(120))
    fully_con_layer = tf.matmul(flatten_layer, w3) + b3
    
    fully_con_layer = tf.nn.relu(fully_con_layer)

    w4 = tf.Variable(tf.truncated_normal(shape=(120,84), mean=mu, stddev=sigma))
    b4 = tf.Variable(tf.zeros(84))
    fully_con_layer = tf.matmul(fully_con_layer, w4) + b4
        
    fully_con_layer = tf.nn.relu(fully_con_layer)

    w5 = tf.Variable(tf.truncated_normal(shape=(84,n_classes), mean = mu, stddev=sigma))
    b5 = tf.Variable(tf.zeros(n_classes))
    logits = tf.matmul(fully_con_layer, w5) + b5
    
    return logits

def evaluate(X_data, y_data):
    num_examples = len(X_data)
    total_accuracy = 0
    sess = tf.get_default_session()
    for offset in range(0, num_examples, BATCH_SIZE):
        batch_x, batch_y = X_data[offset:offset+BATCH_SIZE], y_data[offset:offset+BATCH_SIZE]
        accuracy = sess.run(accuracy_operation, feed_dict={x: batch_x, y: batch_y})
        total_accuracy += (accuracy * len(batch_x))
    return total_accuracy / num_examples

tf.nn.top_k will return the values and indices (class ids) of the top k predictions. So if k=3, for each sign, it'll return the 3 largest probabilities (out of a possible 43) and the correspoding class ids.
